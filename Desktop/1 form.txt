    <script>
    
        document.getElementById("formTabBtn").addEventListener("click", function() {
            document.getElementById("formContainer").classList.add("active");
            document.getElementById("dataTableContainer").classList.remove("active");
            document.getElementById("formTabBtn").classList.add("active");
            document.getElementById("tableTabBtn").classList.remove("active");
        });
 
        document.getElementById("tableTabBtn").addEventListener("click", function() {
            document.getElementById("formContainer").classList.remove("active");
            document.getElementById("dataTableContainer").classList.add("active");
            document.getElementById("tableTabBtn").classList.add("active");
            document.getElementById("formTabBtn").classList.remove("active");
        });
        
        const submitBtn = document.getElementById("submitBtn");
        const resetBtn = document.getElementById("resetBtn");
        const form = document.getElementById("userForm");
 
        function checkFormFilled() {
            let isFormFilled = false;
            form.querySelectorAll('input, select, textarea').forEach(input => {
                if (input.value.trim() !== '') {
                    isFormFilled = true;
                }
            });
            submitBtn.disabled = !isFormFilled;
            resetBtn.disabled = !isFormFilled;
        }
 
        form.addEventListener("input", checkFormFilled);
 
        resetBtn.addEventListener("click", () => {
            form.reset();
            checkFormFilled();
        });
 
        function validateForm() {
            const phone = document.getElementById("phone").value;
            const city = document.getElementById("city").value;
            const state = document.getElementById("state").value; 
            const country = document.getElementById("country").value;
            const firstName = document.getElementById("firstName").value;  
            const email = document.getElementById("email").value;
            const lastName = document.getElementById("lastName").value;
            const company = document.getElementById("company").value;
            const username = document.getElementById("username").value;
            const jobTitle = document.getElementById("jobTitle").value; 
            const website = document.getElementById("website").value; 
            const password = document.getElementById("password").value;
            const confirmPassword = document.getElementById("confirmPassword").value;
  
            if (!/^\d{10}$/.test(phone)) {
                alert("Phone number must be exactly 10 digits.");
                return false;
            }
            if (!/^[A-Za-z]+$/.test(firstName)) {
                alert("First name must contain only letters and be more than 2 characters.");
                return false;
            }
 
            if (!/^[A-Za-z]+@[A-Za-z]+\.[A-Za-z]+$/.test(email)) {
                alert("Email must only contain letters and an '@' symbol.");
                return false;
            }
            if (!/^[A-Za-z\s]+$/.test(username)) {
                alert("Username must only contain letters and spaces.");
                return false;
            }
 
            if (jobTitle && !/^[A-Za-z\s]+$/.test(jobTitle)) {
                alert("Job title must only contain letters and spaces.");
                return false;
            }
            if (website && !/^(https?:\/\/)?([a-zA-Z0-9-]+\.)+[a-zA-Z]{2,6}(\/[a-zA-Z0-9-]*)*$/.test(website)) {
                alert("Please enter a valid website URL starting with 'http://' or 'https://'.");
                return false;
            }
 
            if (company && !/^[A-Za-z\s]+$/.test(company)) {
                alert("Company name must only contain letters and spaces.");
                return false;
            }
 
            if (!/^[A-Za-z]+$/.test(lastName)) {
                alert("Last name must contain only letters and be more than 2 characters.");
                return false;
            }
 
            if (!/^[A-Za-z\s]+$/.test(city)) {
                alert("City name must only contain letters and spaces.");
                return false;
            }
  
            if (!/^[A-Za-z\s]+$/.test(state)) {
                alert("State name must only contain letters and spaces.");
                return false;
            }  
 
            if (!/^[A-Za-z\s]+$/.test(country)) {
                alert("Country name must only contain letters and spaces.");
                return false;
            }
           
            if (password !== confirmPassword) {
                alert("Entered passwords do not match.");
                return false;
            }
 
            return true;
        }
 
         function handleFormSubmission() {
            if (!validateForm()) {
                return;
            }        
 
            const formData = {
                email: document.getElementById("email").value,
                firstName: document.getElementById("firstName").value,
                lastName: document.getElementById("lastName").value,
                address: document.getElementById("address").value,
                phone: document.getElementById("phone").value,
                dob: document.getElementById("dob").value,
                city: document.getElementById("city").value,
                state: document.getElementById("state").value,
                zip: document.getElementById("zip").value,
                country: document.getElementById("country").value,
                username: document.getElementById("username").value,
                password: document.getElementById("password").value,
                company: document.getElementById("company").value,
                jobTitle: document.getElementById("jobTitle").value,
                website: document.getElementById("website").value,
                bio: document.getElementById("bio").value,
                gender: document.getElementById("gender").value,
                interests: document.getElementById("interests").value,
                skills: document.getElementById("skills").value,
                education: document.getElementById("education").value,
                experience: document.getElementById("experience").value
            };
 
            localStorage.setItem("formData", JSON.stringify(formData));
            alert("Form data has been saved!");
            document.getElementById("userForm").reset();
            displayData();
        }
 
        function displayData() {
            const storedData = JSON.parse(localStorage.getItem("formData"));
 
            if (storedData) {
                const tbody = document.getElementById("dataTable").getElementsByTagName('tbody')[0];
                const newRow = tbody.insertRow();
 
                newRow.innerHTML = `
                    <td><input type="checkbox" class="rowCheckbox" /></td>
                    <td>${storedData.email}</td>
                    <td>${storedData.firstName}</td>
                    <td>${storedData.lastName}</td>
                    <td>${storedData.phone}</td>
                    <td>${storedData.dob}</td>
                    <td>${storedData.city}</td> 
                    <td>${storedData.state}</td>
                    <td>${storedData.zip}</td>
                    <td>${storedData.country}</td>
                    <td>${storedData.username}</td>
                    <td>${storedData.company || ''}</td>
                    <td>${storedData.jobTitle || ''}</td>
                    <td>${storedData.website || ''}</td>
                    <td>${storedData.bio || ''}</td>
                    <td>${storedData.gender}</td>
                    <td>${storedData.interests}</td>
                    <td>${storedData.skills}</td>
                    <td>${storedData.education || ''}</td>
                    <td>${storedData.experience}</td>
                    <td><button class="deleteBtn">Delete</button></td>
                    <td><button class="editBtn">Edit</button></td>
                `;
                
                initializeRowButtons(newRow);
            }
        }
        
        function initializeRowButtons(row) {
            const deleteBtn = row.querySelector('.deleteBtn');
            const editBtn = row.querySelector('.editBtn');
            
            deleteBtn.addEventListener('click', function() {
                if (confirm('Are you sure you want to delete this record?')) {
                    row.remove();
                }
            });
            
            editBtn.addEventListener('click', function() {
                handleRowEdit(row, editBtn);
            });
        }
        
        
        const selectAllCheckbox = document.getElementById('selectAll');
        const deleteSelectedBtn = document.getElementById('deleteSelectedBtn');
        
        selectAllCheckbox.addEventListener('change', function() {
            const rowCheckboxes = document.querySelectorAll('.rowCheckbox');
            rowCheckboxes.forEach(function(checkbox) {
                checkbox.checked = selectAllCheckbox.checked;
            });
        });
        
        deleteSelectedBtn.addEventListener('click', function() {
            const rows = document.querySelectorAll('#dataTable tbody tr');
            let hasSelected = false;
            
            rows.forEach(function(row) {
                const checkbox = row.querySelector('.rowCheckbox');
                if (checkbox && checkbox.checked) {
                    hasSelected = true;
                    row.remove();
                }
            });
            
            if (hasSelected) {
                alert('Selected rows have been deleted.');
            } else {
                alert('No rows selected for deletion.');
            }
            
            
            selectAllCheckbox.checked = false;
        });
    
      
        function handleRowEdit(row, editBtn) {
            const cells = row.cells;
            const isEditing = row.classList.contains('editing');
            
            if (isEditing) {       
              
                editBtn.textContent = 'Edit';
                row.classList.remove('editing');
                
              
                for (let i = 1; i < cells.length - 2; i++) {
                    const input = cells[i].querySelector('input, select, textarea');
                    if (input) {
                        cells[i].textContent = input.value;
                    }
                }
            } else {
                
          editBtn.textContent = 'Save';
                row.classList.add('editing');
               
                for (let i = 1; i < cells.length - 2; i++) {
                    const cellText = cells[i].textContent;
                     const fieldName = document.querySelector('#dataTable thead th:nth-child(' + (i + 1) + ')').textContent;
                    
                
                    let inputHTML;
                    if (fieldName === 'Gender') {
                        inputHTML = `<select>
                            <option value="Male" ${cellText === 'Male' ? 'selected' : ''}>Male</option>
                            <option value="Female" ${cellText === 'Female' ? 'selected' : ''}>Female</option>
                            <option value="Other" ${cellText === 'Other' ? 'selected' : ''}>Other</option>
                        </select>`;
                    } else if (fieldName === 'Date of Birth') {
                        inputHTML = `<input type="date" value="${cellText}">`;
                    } else if (fieldName === 'Phone' || fieldName === 'Zip Code' || fieldName === 'Experience') {
                        inputHTML = `<input type="number" value="${cellText}">`;
                    } else if (fieldName === 'Email') {
                        inputHTML = `<input type="email" value="${cellText}">`;
                    } else if (fieldName === 'Website') {
                        inputHTML = `<input type="url" value="${cellText}">`;
                    } else if (fieldName === 'Bio') {
                        inputHTML = `<textarea>${cellText}</textarea>`;
                    } else {
                        inputHTML = `<input type="text" value="${cellText}">`;
                    } 
                    
                    cells[i].innerHTML = inputHTML;
                }
            }
        }
 
    document.querySelectorAll('#dataTable tbody tr').forEach(row => {
            initializeRowButtons(row);
        });
         
 document.getElementById("submitBtn").addEventListener("click", handleFormSubmission);
